using System;
using System.Collections.Generic;
using System.Text;

namespace ConsoleApp2
{
    class TestProgram
    {
        public void callMethods()
        {
            Program obj = new Program();
            obj.Test1();
            obj.Test2(6, 15);
            Console.WriteLine(obj.Test3());
            Console.WriteLine(obj.Test4("nit"));
        }
        static void Main()
        {
            new TestProgram().callMethods();
            Console.ReadLine();
        }
    }
}
