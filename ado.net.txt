using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;
using System.Data.OleDb;

namespace WebApplication2
{
    public partial class WebForm1 : System.Web.UI.Page
    {
        OleDbConnection con;
        OleDbCommand cmd;
        OleDbDataReader dr;
        protected void Page_Load(object sender, EventArgs e)
        {
            con = new OleDbConnection("provider=sqlOleDb; server=.; Integrated Security =SSPI; Database=asp9am"); ;
            cmd = new OleDbCommand("emp_select",con);
            cmd.CommandType = System.Data.CommandType.StoredProcedure;
            con.Open();
            dr = cmd.ExecuteReader();
            if(!IsPostBack)
            {
                ddl1.Items.Insert(0,"select");
                while(dr.Read())
                {
                    ddl1.Items.Add(dr.GetValue(0).ToString());
                }
            }
            con.Close();

        }

        protected void Button1_Click(object sender, EventArgs e)
        {
            int no = int.Parse(ddl1.SelectedItem.Text);
            cmd = new OleDbCommand("emp_selectN",con);
            cmd.Parameters.AddWithValue("@Eno",no);
            cmd.CommandType = System.Data.CommandType.StoredProcedure;
            con.Open();
            dr = cmd.ExecuteReader();
            if(dr.Read())
            {
                txt2.Text = dr.GetValue(0).ToString();
                txt3.Text = dr.GetValue(1).ToString();
            }
            con.Close();
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;
using System.Data.SqlClient;
using System.Data;

namespace WebApplication2
{
    public partial class WebForm2 : System.Web.UI.Page
    {
        SqlConnection con;
        SqlCommand cmd;
        protected void Page_Load(object sender, EventArgs e)
        {
            con = new SqlConnection("server=.; Integrated Security=SSPI; Database=asp9am");
        }

        protected void Button1_Click(object sender, EventArgs e)
        {
            int no = int.Parse(txt1.Text);
            string name = txt2.Text;
            double sal = double.Parse(txt3.Text);
            cmd = new SqlCommand("emp_Insert",con);
            cmd.Parameters.AddWithValue("@eno",no);
            cmd.Parameters.AddWithValue("@ename",name);
            cmd.Parameters.AddWithValue("@Salary",sal);
            cmd.CommandType = System.Data.CommandType.StoredProcedure;
            con.Open();
            int i = cmd.ExecuteNonQuery();
            con.Close();
            txt1.Text = "";
            txt2.Text = "";
            txt3.Text = "";
            if(i>0)
            {
                lbl1.Text = "Record inserted";  
            }
            else
            {
                lbl1.Text = "Record not inserted";
            }
        }
    }
}

        {
            con = new SqlConnection("server=.;Integrated security=SSPI; Database=asp9am");
            cmd = new SqlCommand("emp_select",con);
            cmd.CommandType = System.Data.CommandType.StoredProcedure;

                    ddl1.Items.Add(dr.GetValue(0).ToString());
                }

            con.Close();


        }

        protected void Button1_Click(object sender, EventArgs e)
        {
            try
