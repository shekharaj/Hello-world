 public class Businesslayer
    {
        string strcon = ConfigurationManager.ConnectionStrings["Democonnection"].ToString();
        SqlConnection con;
        SqlCommand cmd;
        // **************** ADD NEW STUDENT *********************
        public bool AddStudent(Student smodel)
        {
          
            SqlCommand cmd = new SqlCommand("GetStudentDetails", con);
            cmd.CommandType = CommandType.StoredProcedure;

            cmd.Parameters.AddWithValue("@SName", smodel.SName);
            cmd.Parameters.AddWithValue("@DOB", smodel.DOB);
            cmd.Parameters.AddWithValue("@DOJ", smodel.DOJ);
            cmd.Parameters.AddWithValue("@FName", smodel.FName);
            cmd.Parameters.AddWithValue("@WClass", smodel.WClass);
            cmd.Parameters.AddWithValue("@Section", smodel.Section);
            cmd.Parameters.AddWithValue("@Address", smodel.Address);
            cmd.Parameters.AddWithValue("@Gender", smodel.Gender);
            cmd.Parameters.AddWithValue("@Fee", smodel.Fee);

            con.Open();
            int i = cmd.ExecuteNonQuery();
            con.Close();

            if (i >= 1)
                return true;
            else
                return false;
        }
       
    }
